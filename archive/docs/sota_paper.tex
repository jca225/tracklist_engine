\title{A Stem-Routed, Indicator-Driven Viterbi Pipeline for DJ-Set Alignment}
\author{Tracklist Engine (\texttt{sota\_v2})}
\date{\today}

\begin{document}
\maketitle

\begin{abstract}
We present \emph{sota\_v2}, the alignment algorithm used by Tracklist Engine to assign every track on a 1001tracklists-scraped tracklist to a mix-side time span on a DJ-set recording. The pipeline (i) routes per-ref MERT comparisons through Demucs stems according to the scraped \texttt{version\_tag}, (ii) infers a monotonic ref-position map with a subsequence-Viterbi DP (replacing argmax, which is non-monotonic and produces impossible descending brackets), (iii) composes an emission score out of three time-series indicators borrowed from technical analysis --- MACD, ATR-normalised, a cross-sectional $z$-score within a variant-group, and a pre-cue persistence baseline --- (iv) runs a per-universe $(K{+}1)$-state Viterbi with a SILENCE state, Phase-5 fingerprint anchors, and a Phase-6 scale-gated cross-universe exclusion mask, and (v) snaps decoded ref endpoints to canonical cue-detr cues computed once per song. On the BB11 ground-truth fixture the pipeline reaches mean mix IoU $0.891$ over the five hand-annotated refs. We report scale failure modes at $119$ refs and the mitigations we added.
\end{abstract}

\section{Problem statement}

Given (a) a DJ-set recording $M$ sampled at $44.1\,\text{kHz}$ with a pre-computed beat-grid $\{t_0, t_1, \dots, t_{T-1}\}$ of measure centres, and (b) a tracklist $\mathcal{R} = \{r_1, \dots, r_N\}$ scraped from 1001tracklists, where each $r \in \mathcal{R}$ is associated with a downloaded reference audio file, its own measure grid, a scraped cue time $c_r$, and a \texttt{version\_tag} indicating which variant the DJ is playing (\emph{acappella}, \emph{instrumental}, or \emph{full}), we seek a mapping
\[
\Phi : \mathcal{R} \to \big(\mathbb{R} \times \mathbb{R} \times \mathbb{R} \times \mathbb{R}\big) \cup \{\bot\}
\]
assigning each ref either a tuple $(s^{\text{mix}}_r, e^{\text{mix}}_r, s^{\text{ref}}_r, e^{\text{ref}}_r)$ describing its mix-side play span and the ref-side region that was played, or $\bot$ (did not play). Alignments are evaluated against hand-annotated ground-truth fixtures using span-IoU.

\section{Notation}

\begin{itemize}[leftmargin=1.5em, itemsep=0.1em]
\item $T$ --- number of mix measures; $N_r$ --- number of measures of ref $r$.
\item $\mathbf{E}^{\text{mix}} \in \mathbb{R}^{T \times d}$, $\mathbf{E}^{r} \in \mathbb{R}^{N_r \times d}$ --- per-measure MERT embeddings on the mix side and the $r$ side respectively.
\item $S_r \in \mathbb{R}^{N_r \times T}$ --- cosine similarity between every ref-measure and every mix-measure of a pairing.
\item $\sigma_r(t) \coloneqq \max_j S_r[j, t]$ --- per-mix-measure max-sim score for ref $r$.
\item $U(r) \in \{\text{acap}, \text{instr}, \text{full}\}$ --- the universe (mutual-exclusion group) of $r$.
\item $\mathcal{U}_u = \{r : U(r) = u\}$ --- the refs in universe $u$.
\end{itemize}

\section{Pipeline overview}

Algorithm~\ref{alg:pipeline} gives the end-to-end flow. Each numbered stage is expanded in the following sections.

\begin{algorithm}[h]
\caption{\textsc{SotaAlignSet}$(\text{set\_id})$}
\label{alg:pipeline}
\begin{algorithmic}[1]
\State $\mathcal{R}, M \leftarrow $ load refs with audio+measures and the set audio
\For{$r \in \mathcal{R}$}
    \State \textbf{stem-route}: pick ref-side and mix-side audio paths per $U(r)$
    \State $\mathbf{E}^r, \mathbf{E}^{\text{mix}} \leftarrow$ MERT-embed per measure (Sec.~\ref{sec:mert})
    \State $S_r \leftarrow \mathbf{E}^r \cdot (\mathbf{E}^{\text{mix}})^\top$  \Comment{cosine similarity}
    \State $\sigma_r \leftarrow \max_\text{axis=0}(S_r)$
    \State $\pi_r \leftarrow \textsc{RefPositionViterbi}(S_r)$ \Comment{Sec.~\ref{sec:refvit}}
\EndFor
\State $A \leftarrow$ \textsc{LoadFingerprintAnchors}(set\_id) \Comment{Phase 5}
\State $X \leftarrow$ \textsc{BuildFullExclusionMask}($A$, $\mathcal{R}$) with coverage cap \Comment{Phase 6}
\For{$u \in \{\text{acap, instr, full}\}$}
    \State $P_u \leftarrow \textsc{ViterbiUniverse}(\mathcal{U}_u, \sigma, A, X_{u \ne \text{full}})$ \Comment{Sec.~\ref{sec:vitu}}
    \State $P_u \leftarrow \textsc{CleanPath}(P_u)$ \Comment{Sec.~\ref{sec:clean}}
\EndFor
\For{$r \in \mathcal{R}$}
    \State extract first run $(s, e)$ of $r$ in $P_{U(r)}$; read endpoints via $\pi_r$
    \State snap $(s^{\text{ref}}_r, e^{\text{ref}}_r)$ to nearest canonical cue; apply mix-side shift
\EndFor
\State persist to \texttt{set\_section\_alignment} with \texttt{confidence\_source='sota\_v2'}
\end{algorithmic}
\end{algorithm}

\section{Signal extraction}
\label{sec:mert}

\subsection{Stem routing}

Let $\text{stem}(u)$ denote the audio stream a ref in universe $u$ should be compared against:
\[
\text{stem}(u) = \begin{cases}
\text{vocals} & u = \text{acap} \\
\text{instrumental} & u = \text{instr} \\
\text{full audio} & u = \text{full}
\end{cases}
\]
We apply the \emph{same} $\text{stem}(u)$ on both sides of the pairing: the mix is demucs-separated once per set and the relevant stem embedded, and each ref is embedded from its own corresponding stem. Routing an acappella-tagged ref against the mix's vocals stem suppresses the instrumental bed that would otherwise dominate MERT's cosine similarity.

\subsection{MERT embeddings and cosine similarity}

Each stem-routed audio file is segmented according to its measure grid; the MERT~v1 encoder produces a $d$-dimensional embedding per measure after mean-pooling across the token axis. We use the layer-5 average as a fixed feature space (determined to be the most alignment-informative on a held-out subset). Similarity between ref-measure $j$ and mix-measure $t$ is the cosine
\[
S_r[j, t] = \frac{\mathbf{E}^r_j \cdot \mathbf{E}^{\text{mix}}_t}{\|\mathbf{E}^r_j\| \, \|\mathbf{E}^{\text{mix}}_t\|}.
\]

\section{Ref-position Viterbi}
\label{sec:refvit}

Given $S_r$, we want a monotonic map $\pi_r : \{0, \dots, T{-}1\} \to \{0, \dots, N_r{-}1\}$ answering \emph{where in ref $r$ is the DJ at mix time $t$?}. Argmax over rows violates monotonicity --- identical choruses at different ref times confuse it --- and produces the pathological brackets $[s^{\text{ref}} > e^{\text{ref}}]$ we observed pre-Viterbi. We instead solve a subsequence-DTW-style shortest-path problem on a $T \times N_r$ cost lattice:

\begin{equation}
\mathcal{C}_t[j] = (1 - S_r[j, t]) + \min_{j'} \Big( \mathcal{C}_{t-1}[j'] + \tau(j - j') \Big),
\end{equation}
\begin{equation}
\tau(\delta) = \begin{cases}
0 & \delta \in \{0, 1\} \quad \text{(stay or advance)} \\
0.3\,\delta & \delta \geq 2 \quad \text{(speed-up skip)} \\
2.0\,|\delta| & \delta < 0 \quad \text{(loop-back)}
\end{cases}
\end{equation}

The large backward cost forbids path descent in almost all cases, eliminating the invalid descending-bracket failure mode. The optimisation is restricted to a band $|j - j'| \le 8$ for compute. The backpointer lattice $\mathcal{B}$ is recovered in $O(T)$ to yield $\pi_r$.

\begin{algorithm}[h]
\caption{\textsc{RefPositionViterbi}$(S_r)$}
\begin{algorithmic}[1]
\State $\mathcal{C}_0[j] \leftarrow 1 - S_r[j, 0]$ for all $j$
\For{$t = 1, \dots, T-1$}
    \For{$j = 0, \dots, N_r-1$}
        \State $(\mathcal{C}_t[j], \mathcal{B}_t[j]) \leftarrow \min_{|j' - j| \le 8} \big(\mathcal{C}_{t-1}[j'] + \tau(j - j')\big) + (1 - S_r[j,t])$
    \EndFor
\EndFor
\State Backtrace from $\arg\min_j \mathcal{C}_{T-1}[j]$ through $\mathcal{B}$ to recover $\pi_r$
\end{algorithmic}
\end{algorithm}

\section{Emission signals: time-series indicators as alignment features}

Rather than consume $\sigma_r$ directly as a score, we compose it with three orthogonal signals borrowed from technical analysis of financial time-series. Each captures a distinct notion of evidence that ref $r$ is currently playing.

\subsection{MACD histogram of similarity momentum}

Let $\text{EMA}_n(x)$ denote the exponential moving average with span $n$. The MACD of $\sigma_r$ is
\[
\text{MACD}_r(t) = \text{EMA}_{12}(\sigma_r)(t) - \text{EMA}_{26}(\sigma_r)(t),
\]
the signal line is $\text{Signal}_r(t) = \text{EMA}_9(\text{MACD}_r)(t)$, and the \emph{histogram}
\[
h_r(t) = \text{MACD}_r(t) - \text{Signal}_r(t)
\]
fires sharply positive at \emph{entries} (similarity rises faster than its recent trend) and negative at \emph{exits} (the ref stops). It is the most informative signal for boundary detection.

To put different refs on a comparable scale we divide by the Wilder-smoothed average true range of $\sigma_r$, $\text{ATR}_{14}(\sigma_r)$, and clip:
\[
\tilde h_r(t) = \text{clip}\!\left(\frac{h_r(t)}{\max(\text{ATR}_{14}(\sigma_r)(t), 10^{-4})}, -3, 3\right).
\]

\subsection{Cross-sectional $z$-score within universe}

At each mix measure $t$ we ask: of all refs in the same mutual-exclusion group $\mathcal{U}_u$, which is the \emph{relative} winner?
\[
\mu_u(t) = \frac{1}{|\mathcal{U}_u|}\sum_{r \in \mathcal{U}_u}\sigma_r(t), \quad
s_u(t) = \sqrt{\tfrac{1}{|\mathcal{U}_u|}\sum_{r \in \mathcal{U}_u}(\sigma_r(t) - \mu_u(t))^2},
\]
\[
z_r(t) = \frac{\sigma_r(t) - \mu_u(t)}{\max(s_u(t), 10^{-6})}, \quad r \in \mathcal{U}_u.
\]
This is the ``max-of-acappellas at each time'' idea formalised: picking the acappella with the strongest relative evidence. For singleton universes ($|\mathcal{U}_u| = 1$) we fall back to a rolling temporal $z$-score of $\sigma_r$ against its own 40-measure history, because a ref with no in-universe competitors would otherwise have a constant-zero $z$ and lose every winner-takes-all contest against SILENCE.

\subsection{Pre-cue persistence baseline}

Because the DJ's scraped cue $c_r$ is an approximate left-bound, we estimate ref $r$'s \emph{silence baseline} as the mean similarity before the cue:
\[
b_r = \text{mean}_{\{t : t < c_r\}}\sigma_r(t),
\]
falling back to the $10^\text{th}$ percentile of $\sigma_r$ if fewer than three pre-cue measures exist. The persistence score
\[
p_r(t) = \text{clip}\!\big(8 \cdot (\sigma_r(t) - b_r),\, -2,\, 2\big)
\]
stays positive for the \emph{full} duration of a play (unlike MACD's histogram, which fires only at entries/exits), holding the Viterbi in-state during long sustained plays.

\subsection{Composite emission score}

The per-ref emission signal consumed by the per-universe Viterbi is a weighted sum:
\begin{equation}
\mathcal{E}_r(t) = 0.5\,z_r(t) + 1.0\,\tilde h_r(t) + 1.5\,p_r(t).
\end{equation}
The weights were tuned on BB11 ground-truth and not re-tuned since.

\section{Fingerprint anchors (Phase 5)}

We pre-compute chromaprint fingerprints for every track and scan the mix against them. A fingerprint hit is a tuple $(t^{\text{mix}}_\text{start}, t^{\text{mix}}_\text{end}, \text{score}, \text{matched\_track\_id})$. Single hits are noisy (a lone 0.7-score hit is nearly meaningless), but \emph{clusters} of hits within a small window are strong positive evidence. For each ref $r$ with track\_id $\tau$ we build a boolean anchor mask $\alpha_r \in \{0,1\}^T$:
\[
\alpha_r(t) = \mathbb{1}\!\left[\,\#\{\text{hits of }\tau \text{ with centre within } \pm10\,\text{s of } t_t,\ \text{score} \ge 0.65\} \ge 2\,\right].
\]
Where $\alpha_r(t) = 1$, the per-universe Viterbi receives a $-1.5$ emission-cost bonus for state $r$, reinforcing the decode at density-confirmed measures.

\section{Full-track exclusion with scale-aware cap (Phase 6)}
\label{sec:phase6}

When a \emph{full}-variant ref is fingerprint-confirmed, layering is unlikely: the DJ has swapped to playing a whole track, not overlaying an acappella on top. We construct a cross-universe exclusion mask
\[
X(t) = \bigvee_{r : U(r) = \text{full}} \alpha_r(t).
\]
At measures where $X(t) = 1$, the emission cost for every non-silence state in the \emph{acappella} and \emph{instrumental} universes is set to $10^6$, forcing those universes to SILENCE.

\paragraph{Scale failure mode.} At small $|\mathcal{R}|$ the union $X$ covers a small fraction of the mix ($\approx 15\%$ on BB11's 5-ref GT). At $|\mathcal{R}| = 119$ we observed $72\%$ coverage, hard-masking genuine acappella plays and regressing Gnash's IoU from $0.857$ to $0.27$.

\paragraph{Mitigation.} We apply $X$ only if
\[
\frac{1}{T}\sum_t X(t) \leq 0.40.
\]
Above the cap, $X$ is discarded and the within-universe Viterbi is allowed to decide. The cap preserves the validated 5-ref behaviour while avoiding the hard-mask regression at scale.

\section{Per-universe K+1-state Viterbi}
\label{sec:vitu}

Within each universe $u$ we decode a Viterbi with state space $\{0, 1, \dots, |\mathcal{U}_u|{-}1\} \cup \{\texttt{SIL}\}$ (one state per ref in the universe plus a single silence state). Emission cost for ref state $i$ at mix measure $t$ is
\[
\mathcal{E}^{\text{cost}}_i(t) = -\mathcal{E}_{r_i}(t) - 1.5 \cdot \alpha_{r_i}(t) + 10^6 \cdot \mathbb{1}[X(t)\wedge u \ne \text{full}],
\]
and SILENCE is constant at $-1.2$. We apply a hard left-boundary: $\mathcal{E}^{\text{cost}}_i(t) = +\infty$ for $t$ such that $t_t < c_{r_i}$. Transition costs are
\begin{center}
\begin{tabular}{@{}lcccc@{}}
\toprule
& \multicolumn{4}{c}{from} \\
\cmidrule{2-5}
to & $i$ & $j \ne i$ & \texttt{SIL} & (note) \\
\midrule
$i$ (self-loop) & $0$ & $3.0$ & $0.4$ & enter-from-silence is $0.4$ \\
$j \ne i$ & $3.0$ & --- & $0.4$ & cross-ref forced via SILENCE \\
\texttt{SIL} & $0.1$ & $0.1$ & $0$ & exit-to-silence is $0.1$ \\
\bottomrule
\end{tabular}
\end{center}
The $3.0$ cross-ref cost makes direct ref-to-ref transitions strictly worse than going through SILENCE, producing at most one active ref per universe per measure. The full Viterbi forward pass is $O(T \cdot (|\mathcal{U}_u|+1)^2)$.

\section{Path cleanup}
\label{sec:clean}

Raw Viterbi paths are fragmented by short incidental SILENCE wins within real plays (1--3 measures). Cleanup per ref $i$ performs:
\begin{enumerate}[leftmargin=1.5em, itemsep=0.1em]
\item \textbf{Merge.} Extract all contiguous runs of state $i$. Merge two consecutive runs iff the gap is $\le 10$ measures \emph{and} every cell in the gap is SILENCE.
\item \textbf{Cue-anchored selection.} Pick the \emph{earliest} merged run of duration $\ge 5$ measures whose start is within $\pm 80\,\text{s}$ of the scraped cue $c_{r_i}$. All other runs (later re-entries, spurious short fires, far-from-cue) are wiped to SILENCE.
\item \textbf{Drop.} If no run qualifies, ref $i$ produces $\Phi(r_i) = \bot$.
\end{enumerate}

The earliest-near-cue rule encodes the DJ-play prior: each ref is played once, starting near its scraped cue. Later re-entries are almost always ref-collision artefacts (a competing ref's signal leaking into this ref's state).

\section{Endpoint extraction and canonical-cue snap}

From the cleaned run $(s, e)$ of ref $r$ we read the ref-position Viterbi path $\pi_r$ at both boundaries. To tolerate single-measure noise we take the median of the first and last three measures of $\pi_r$ within the run:
\[
j^{\text{lo}}_r = \text{median}\big(\pi_r[s\,{:}\,s{+}3]\big), \quad j^{\text{hi}}_r = \text{median}\big(\pi_r[e{-}3\,{:}\,e]\big).
\]
These map to ref-time seconds $t^{\text{ref,lo}}_r$, $t^{\text{ref,hi}}_r$ via the ref-side measure centres.

\paragraph{Canonical cue snap.} Each song has a canonical cue list $\mathcal{K}_\tau$ computed \emph{once} on its original \texttt{variant\_tag='original'} full-song audio at cue-detr sensitivity $0.5$, stored in \texttt{canonical\_track\_cue\_points} keyed by \texttt{track\_id}. All variants (acappella, instrumental, full, remix) share the same cue list because the underlying song structure is the same. We snap
\[
\tilde t^{\text{ref,lo}}_r = \arg\min_{k \in \mathcal{K}_\tau} |k - t^{\text{ref,lo}}_r|, \quad \tilde t^{\text{ref,hi}}_r = \arg\min_{k \in \mathcal{K}_\tau} |k - t^{\text{ref,hi}}_r|.
\]
Assuming a tempo factor of $\approx 1$, the mix-side span is shifted by the same delta:
\[
\tilde s^{\text{mix}}_r = s^{\text{mix}}_r - (t^{\text{ref,lo}}_r - \tilde t^{\text{ref,lo}}_r), \quad
\tilde e^{\text{mix}}_r = e^{\text{mix}}_r - (t^{\text{ref,hi}}_r - \tilde t^{\text{ref,hi}}_r).
\]
We \emph{skip} the snap for $u = \text{full}$ refs, because cue-detr on a full-band track fires a lot and the nearest cue frequently lies well inside the real play span, regressing the outer-boundary IoU (verified: snapping Antoine on BB11 regressed its IoU from $0.950$ to $0.826$).

\section{Evaluation}

\subsection{Ground-truth fixture}

We annotate a DJ-set against a user-maintained YAML fixture at \texttt{tests/fixtures/<set\_id>\_ground\_truth.yaml} with hand-verified $(s^{\text{mix}}, e^{\text{mix}})$ for every played track, plus optional \texttt{ref\_segments} for loops and cut-ups. The primary mix-side metric is span-IoU
\[
\text{IoU}_r = \frac{|[\tilde s^{\text{mix}}_r, \tilde e^{\text{mix}}_r] \cap [s^{\text{mix,gt}}_r, e^{\text{mix,gt}}_r]|}{|[\tilde s^{\text{mix}}_r, \tilde e^{\text{mix}}_r] \cup [s^{\text{mix,gt}}_r, e^{\text{mix,gt}}_r]|}.
\]

\section{Discussion: scale failure modes}

Applying \emph{sota\_v2} to the full BB11 tracklist ($119$ refs, vs. the $5$-ref GT subset) dropped mean IoU to $0.629$ before mitigation. We identified three failure modes, in descending order of impact:

\begin{enumerate}[leftmargin=1.5em, itemsep=0.2em]
\item \textbf{$X$ over-masking (dominant).} The full-track exclusion union grows with $|\mathcal{U}_\text{full}|$; at $119$ refs it covered $72\%$ of the mix, hard-forcing acappella and instrumental universes to SILENCE across genuine play windows. \emph{Mitigation}: the coverage cap in Sec.~\ref{sec:phase6} --- if $\frac{1}{T}\sum_t X(t) > 0.40$, disable $X$ entirely for the set.
\item \textbf{Version-tag mis-classification.} \texttt{\_normalise\_version\_tag} is a keyword regex; refs whose scraped metadata lacks ``acap'', ``instr'', or ``dub'' default to \emph{full}, over-populating $\mathcal{U}_\text{full}$ and feeding (1). For the GT-validated BB11 subset we maintain a hand-curated override map.
\item \textbf{Cross-sectional $z$ dilution.} $\mu_u, s_u$ are computed over the full universe. With $|\mathcal{U}_u| \approx 40$ per universe the signal-to-noise of each individual $z_r$ degrades. An open direction is a rolling temporal $z$ per ref instead of (or combined with) the cross-sectional one.
\end{enumerate}

\section{Open problems}

\begin{enumerate}[leftmargin=1.5em, itemsep=0.1em]
\item \textbf{Version-tag audio-level verification.} Replace the scraped-metadata regex with a classifier that listens to the ref audio (vocals-to-total energy ratio, duration) and confirms the variant.
\item \textbf{Scale-aware $z$.} Rolling temporal $z$ or robust location/scale (median / MAD) per ref rather than pooled cross-sectional moments, to stabilise emission signals at large $|\mathcal{U}_u|$.
\item \textbf{Soft full-exclusion.} Replace the hard $10^6$ mask with a continuous penalty weighted by the density / confidence of the triggering fingerprint anchor --- recovers (1)'s benefit at small scale without the pathological behaviour at large scale.
\item \textbf{Loops and cut-ups.} The current pipeline emits a single $(s^{\text{ref}}, e^{\text{ref}})$ bracket per ref. GT fixtures admit \texttt{ref\_segments} for loops; decoding the ref-position Viterbi into multiple runs and reporting each is the natural extension.
\end{enumerate}
